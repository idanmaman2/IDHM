def isprime3(r : int )-> (bool , tuple[tuple[int]]):
    total =  defaultdict(int)
    isprime = True
    def iterate_prime(prime:int)->(int,bool):
        isprime2,z = isprime ,r
        while not(z % prime) :
            isprime2 = False
            total[prime] += 1
            z = z// prime
        return z, isprime2
    r,isprime=iterate_prime(2)

        total[r] = 1
    return isprime,  tuple(total.items())
